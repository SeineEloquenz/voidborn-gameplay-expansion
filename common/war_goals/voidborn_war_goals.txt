wg_void_assimilation = {
	casus_belli = cb_void_assimilation
    
    icon = GFX_wg_void_void_assimilation
	
	hide_if_no_cb = no
	total_war = no
	
	surrender_acceptance = -1000

	potential = {
		from = {
            OR = {
                is_species_class = HUM
                is_species_class = MAM
                is_species_class = CORRUPTED
            }
        }
		has_valid_civic = civic_void_assimilation
	}
	possible = {
		always = yes
	}
    
    on_accept = {
		custom_tooltip = wg_void_assimilation_ttp
        hidden_effect = {
            save_event_target_as = vb_assimilator
            owner_species = {
                save_event_target_as = species_scope
            }
            from = {
                country_event = { id = voidborn.11100}
            }
        }
    }
	
	ai_weight = {
		weight = 1000
	}
}

    icon = GFX_wg_void_void_abduct
	
	hide_if_no_cb = no
	total_war = no
	
	surrender_acceptance = -25

	potential = {
		from = {
            OR = {
                is_species_class = HUM
                is_species_class = MAM
                is_species_class = CORRUPTED
            }
        }
		has_valid_civic = civic_void_assimilation
	}
    
	possible = {
		always = yes
	}
    
	on_accept = {
		custom_tooltip = wg_abduct_accept			
        hidden_effect = { 
            from = {
                every_owned_pop = {
                    random_list = {
                        25 = {
                            save_event_target_as = abduct_pop
                            root = {
                                random_owned_planet = {
                                    resettle_pop = {
                                        pop = event_target:abduct_pop
                                        planet = this
                                    }
                                }
                            }
                        }
                        75 = {}
                    }
				}
            add_modifier = { modifier = raided days = 3600 } 
            }
        }
    }
	
	ai_weight = {
		weight = 1000
	}
}
